    <script>
        function goToStudent(studentName) {
            // In a real app, this would navigate to the student's dashboard
            console.log(`Navigating to ${studentName}'s dashboard`);
            // window.location.href = `/student/${studentName}`;
        }

        // Add click handlers for cards
        document.querySelectorAll('.student-card').forEach(card => {
            card.addEventListener('click', function() {
                const link = this.querySelector('.start-button');
                if (link) {
                    window.location.href = link.href;
                }
            });
        });

        // Prevent double navigation when clicking the button directly
        document.querySelectorAll('.start-button').forEach(button => {
            button.addEventListener('click', function(e) {
                e.stopPropagation();
            });
        });

        // Add subtle animations
        const observerOptions = {
            threshold: 0.1,
            rootMargin: '0px 0px -50px 0px'
        };

        const observer = new IntersectionObserver(function(entries) {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.style.opacity = '1';
                    entry.target.style.transform = 'translateY(0)';
                }
            });
        }, observerOptions);

        // Animate cards on load
        document.querySelectorAll('.student-card').forEach((card, index) => {
            card.style.opacity = '0';
            card.style.transform = 'translateY(20px)';
            card.style.transition = 'opacity 0.6s ease, transform 0.6s ease';
            card.style.transitionDelay = `${index * 0.1}s`;
            observer.observe(card);
        });
    </script>
