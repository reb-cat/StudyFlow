    <script>
        function toggleAdvanced() {
            const content = document.getElementById('advanced-content');
            const text = document.getElementById('advanced-text');
            const icon = document.getElementById('advanced-icon');
            
            if (content.classList.contains('show')) {
                content.classList.remove('show');
                text.textContent = 'Show More Filters';
                icon.style.transform = 'rotate(0deg)';
            } else {
                content.classList.add('show');
                text.textContent = 'Hide Filters';
                icon.style.transform = 'rotate(180deg)';
            }
        }

        function markSelectedComplete() {
            const checkboxes = document.querySelectorAll('.assignment-checkbox:checked');
            alert(`Would mark ${checkboxes.length} assignments as complete`);
        }
    </script>
